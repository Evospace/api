33bool UMod::DeserializeFromDirectory(const FString &directory, ModLoadingContext &context) {
34 IPlatformFile &PlatformFile = FPlatformFileManager::Get().GetPlatformFile();
35 {
36 FString json_data;
37 if (PlatformFile.FileExists(*(directory / "info.json"))) {
38 FFileHelper::LoadFileToString(json_data, *(directory / "info.json"));
39 } else {
40 LOG(WARN_LL) << *directory << " has no info.json";
41 return false;
42 }
43
44 TSharedPtr<FJsonObject> JsonObject = MakeShareable(new FJsonObject());
45 TSharedRef<TJsonReader<>> JsonReader = TJsonReaderFactory<>::Create(json_data);
46
47 if (FJsonSerializer::Deserialize(JsonReader, JsonObject) && JsonObject.IsValid()) {
48 DeserializeJson(JsonObject);
49 }
50 }
51
52 auto lua_state = context.lua_state;
53
54 auto file = directory / "init.lua";
55 if (PlatformFile.FileExists(*file)) {
56 FString json_data;
57 FFileHelper::LoadFileToString(json_data, *(directory / "init.lua"));
58 luabridge::setGlobal(lua_state->L, nullptr, "init");
59 context.loader->lastRegisteredMod = {};
60 if (auto result = lua_state->RunCode(TCHAR_TO_UTF8(*json_data), "@init")) {
61 if (context.loader->lastRegisteredMod.has_value()) {
62 mod_table = context.loader->lastRegisteredMod.value();
63 if (this->mod_table->isTable()) {
64 LOG(INFO_LL) << "Init function table registered";
65 } else {
66 LOG(ERROR_LL) << "db:mod expects to get a table";
67 }
68 } else {
69 LOG(ERROR_LL) << "Registration unknown error";
70 }
71 } else {
72 LOG(ERROR_LL) << "init.lua file execution error";
73 }
74 }
75
76 return true;
77}
