13template <typename T>
14void RegisterClassAndSuper(UClass *Class, TSet<UClass *> &Registered, lua_State *L) {
15 if (!Class || Registered.Contains(Class)) {
16 return;
17 }
18
19 // Register superclass first
20 auto super = Class->GetSuperClass();
21 if (super && super != Class && super->IsChildOf(T::StaticClass())) {
22 RegisterClassAndSuper<T>(Class->GetSuperClass(), Registered, L);
23 }
24
25 if (const auto cdo = Class->GetDefaultObject<T>()) {
26 if (!Registered.Contains(cdo->lua_reg_type())) {
27 cdo->register_owner();
28 cdo->lua_reg(L);
29 cdo->lua_reg_internal(L);
30 Registered.Add(Class);
31 }
32 }
33}

35void ModLoadingLuaState::Init() {
36 LuaState::Init();
37
38 TSet<UClass *> registered = {};
39 for (TObjectIterator<UClass> it; it; ++it) {
40 if (it->IsChildOf(UPrototype::StaticClass()) && !registered.Contains(*it)) {
41 RegisterClassAndSuper<UPrototype>(*it, registered, L);
42 } else if (it->IsChildOf(UInstance::StaticClass()) && !registered.Contains(*it)) {
43 RegisterClassAndSuper<UInstance>(*it, registered, L);
44 }
45 }
46}
